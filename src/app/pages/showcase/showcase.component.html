Breadcrumbs

API Reference

API Reference

variant'line' | 'enclosed' | 'pills' | 'soft''line'Visual style variant
OutputTypeDescription
tabChangeEventEmitter<string>Emits the tab ID when a tab is selected

@if (isModalOpen()) {
  <ui-modal
    [isOpen]="isModalOpen()"
    [size]="modalSize"
    title="Example Modal"
    [showCloseButton]="true"
    (close)="closeModal()">
    <p>Modal content goes here.</p>

// In component.ts:
isModalOpen = signal(false);
modalSize: 'sm' | 'md' | 'lg' | 'xl' = 'md';

openModal(size: 'sm' | 'md' | 'lg') {
  this.modalSize = size;
  this.isModalOpen.set(true);
}

closeModal() {
  this.isModalOpen.set(false);
}

  this.toastColor = color;
  this.showToast.set(true);
}

hideToast() {
  this.showToast.set(false);
}
